Loops

for Most commonly used loop in JavaScript Example 
let theBest = ['Kirtley', 'Captain', 'Hero'];
for (let i = 0; i < theBest.length; i++) {
  console.log(theBest[i]);
};
             
// Kirtley
// Captain
// Hero
while Sets conditions in which a loop will execute Example 
let count = 0;
while (count < 5) {
  console.log(`Current count is ${count}`);
  count++;
}

// Current count is 0
// Current count is 1
// Current count is 2
do while Similar to while loop - Will execute at least one time and checks to see if conditions are met to execute again or exit Example 
let count = 0;
do {
  console.log(`Current count is ${count}`);
  count++;
} while (count < 3);

// Current count is 0
// Current count is 1
// Current count is 2
forEach For loop that will execute for every element in an array Example 
let theBest = ['Kirtley', 'Captain', 'Hero'];
theBest.forEach(element => {
  console.log(element);
});

// Kirtley
// Captain
// Hero

Array Methods

pop() Remove last element from an array Example 
let array = ['Kitley', 'Wants', 'Better', 'Icecream'];
array.pop();
array;
  ['Kitley', 'Wants', 'Better']
push() Add a new element to the end of an array Example 
let array = ['Kitley', 'Says', 'This', 'Sucks'];
array.pop('Hard');
array;
  ['Kitley', 'Says', 'This', 'Sucks', 'Hard']
shift() Removes the first element of an array Example 
let array = ['Kitley', 'Knits', 'Cute', 'Socks'];
array.shift();
array;
  ['Knits', 'Cute', 'Socks']
unshift() Add a new element to the beginning of an array Example 
let array = ['Kitley', 'Owns', 'This', 'House'];
array.unshift('Captain');
array;
  ['Captain', 'Kirtley', 'Owns', 'This', 'House']
split() Creates a new array from a string divided by a specified value Example 
let truth = 'Kitley Loves Puppy Cuddles';
let array = truth.split(" ");
array;
  ['Kirtley', 'Loves', 'Puppy', 'Cuddles']
join() Combine all elements in an array and return a string Example 
let array = ['Kitley', 'Wants', 'All', 'The', 'Food'];
array.join(" ");
array;
  'Kirtley Wants All The Food'
concat() Join multiple arrays into one Example 
let array1 = ['Kitley', 'Is', 'Amazing'];
let array2 = ['She', 'Is', 'Loved', 'By', 'All'];
let array3 = array1.concat(array2);
array3;
  ['Kitley', 'Is', 'Amazing', 'She', 'Is', 'Loved', 'By', 'All']
reverse() Reverse the order of all elements in an arrays Example 
let array = ['Kitley', 'Will', 'Eat', 'Godzilla'];
array.reverse();
array;
  ['Godzilla', 'Eat', 'Will', 'Kirtley']
sort() Sorts all elements in an array alphabetically Example 
let array = ['Kitley', 'Favorite', 'Animal', 'Is', 'Sloths'];
array.sort();
array;
  ['Animal', 'Favorite', 'Is', 'Kitley', 'Sloths']
indexOf() Returns the position of the first element in an array Example 
let array = ['the', 'duck', 'crossed', 'the', 'road', 'with', 'Kirtley', 'such', 'a', 'cute', 'duck'];
array.indexOf('duck');
  1
slice() Removes specified elements of a selected string. Start parameter is required, end parameter is optional Example 
let str = 'Abandoned'
str.slice(7);     // ed
str.slice(-2);    // ned
str.slice(2,5);   // and
str.slice(3,-1);  // ndone
str.slice(0);     // Abandoned

Arrow Functions

Arrow functions are a shorthand way to write functions. Using them you can remove {} as well as return in several cases
Basic Arrow Function Example 
// ORIGINAL
function peanut(num) {
  return num*3;
}

// ARROW 
let peanut = num => num*3;
Arrow Function with shot hand if statement Example 
// ORIGINAL
function butter(n1, n2) {
  if (n1 === n2) {
    return 'Delicous';
  } else {
    return 'Gross!'
  }
}

// ARROW
let butter = (n1, n2) => n1 === n2 ? 'Delicous' : 'Gross'
